77def match_file_in_folder(folder, pattern): 

78 """Search for files in the specified folder by pattern. If any 

79 found, return a full path to the file. Otherwise, raise 

80 'FileNotFoundError' error. 

81 

82 Args: 

83 folder (str): An absolute path to the folder. 

84 pattern (str): Unix shell-style wildcards. 

85 

86 Examples: 

87 >>> match_file_in_folder('/home/user/bin', '*main.dfw') 

88 '/home/user/bin/internal_ramdrive_main.dfw' 

89 """ 

90 file_path = None 

91 for file_name in os.listdir(folder): 

92 if fnmatch.fnmatch(file_name, pattern): 

93 file_path = os.path.join(folder, file_name)

95 raise FileNotFoundError(f"File '{pattern}' not found in the '{folder}' directory.")
